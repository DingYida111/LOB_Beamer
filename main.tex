\section{LOB Basics}
\begin{frame}[plain]
\contentheader{The Object: A Queue-Shaped State}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.52\textwidth}
\begin{itemize}\tightitems
  \item A LOB records resting buy and sell orders that have not yet traded.
  \item Each level has a price and a queue size: best ask, second ask, best bid, second bid, and so on.
  \item Deep LOB models usually keep the first \(L=10\) levels and write one snapshot as a vector.
\end{itemize}
\vspace{0.3em}
\begin{block}{One Snapshot}
\[
L(\tau)=
\{p^{ask}_{\ell},v^{ask}_{\ell},p^{bid}_{\ell},v^{bid}_{\ell}\}_{\ell=1}^{L}
\in\mathbb{R}^{4L}.
\]
\end{block}
\end{column}
\begin{column}{0.44\textwidth}
\begin{block}{The Model's Observation}
\[
X_\tau =
\big(L(\tau-T+1),\ldots,L(\tau)\big)
\]
\[
X_\tau\in\mathbb{R}^{T\times4L}
\]
\end{block}
\begin{block}{What to Picture}
The input is not a price line. It is a short movie of queue shapes on both sides of the market.
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[plain]
\contentheader{A Queueing Model of the Book}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.50\textwidth}
\begin{itemize}\tightitems
  \item A useful abstraction is a continuous-time state process:
  \[
    Q_t=(Q_t^{b,1},\ldots,Q_t^{b,L},Q_t^{a,1},\ldots,Q_t^{a,L}).
  \]
  \item Each coordinate is queue size at one bid/ask level.
  \item Limit orders, market orders, and cancellations are jumps of \(Q_t\).
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\begin{block}{Point-process view}
\[
dQ_t=\sum_{e\in\mathcal{E}} \nu_e(Q_{t-})\,dN_t^e
\]
\[
\mathbb{P}(dN_t^e=1\mid\mathcal{F}_{t-})
\approx \lambda_e(Q_{t-})\,dt
\]
\end{block}
\begin{block}{Mathematical Meaning}
The book is not a smooth curve; it is a queueing system hit by random order-flow events.
\end{block}
\end{column}
\end{columns}
\source{Cont and de Larrard (2013); Abergel and Jedidi (2013).}
\end{frame}

\begin{frame}[plain]
\contentheader{Price Changes as Queue Depletion}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\begin{block}{Price moves when a queue is depleted}
\scriptsize
\[
\tau_a=\inf\{t:Q_t^{a,1}=0\}
\]
\[
\tau_b=\inf\{t:Q_t^{b,1}=0\}
\]
\[
\text{next move}=
\begin{cases}
\text{up}, & \tau_a<\tau_b,\\
\text{down}, & \tau_b<\tau_a.
\end{cases}
\]
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{itemize}\tightitems
  \item Queue imbalance is predictive because it changes depletion probabilities.
  \[
    I_t=\frac{Q_t^{b,1}-Q_t^{a,1}}{Q_t^{b,1}+Q_t^{a,1}}.
  \]
  \item Event clustering matters: order arrivals are often self- and cross-exciting.
  \[
    \lambda_i(t)=\mu_i+\sum_j\int_0^t \phi_{ij}(t-s)\,dN_j(s).
  \]
  \item Forecasting asks for conditional probabilities such as
  \[
    \mathbb{P}(\tau_a<\tau_b\mid\mathcal{F}_t).
  \]
\end{itemize}
\end{column}
\end{columns}
\source{Cont and de Larrard (2013); Bacry, Mastromatteo and Muzy (2015).}
\end{frame}

\begin{frame}[plain]
\contentheader{From Queue Dynamics to a Label}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\begin{block}{Mid-Price and Spread}
\[
m_\tau=\frac{p^{ask}_{1}(\tau)+p^{bid}_{1}(\tau)}{2}
\]
\[
\sigma_\tau=p^{ask}_{1}(\tau)-p^{bid}_{1}(\tau)
\]
\end{block}
\begin{itemize}\tightitems
  \item \(m_\tau\): the reference price used to define direction.
  \item \(\sigma_\tau\): the bid--ask spread, also a direct trading friction.
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Three-Class Label}
\[
y_\tau =
\begin{cases}
-1, & m_{\tau+\Delta\tau}-m_\tau\le -\theta,\\
0, & |m_{\tau+\Delta\tau}-m_\tau|<\theta,\\
+1, & m_{\tau+\Delta\tau}-m_\tau\ge +\theta.
\end{cases}
\]
\end{block}
\begin{itemize}\tightitems
  \item \(\theta\): tick size, the smallest admissible price change.
  \item \(\Delta\tau\): a horizon in future LOB updates, not in fixed seconds.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[plain]
\contentheader{Where Analytic Models Break}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\begin{block}{The classical route}
\[
\mathbb{P}(y_\tau=c\mid X_\tau)
\]
can be derived only after strong assumptions on event intensities, independence, stationarity, and low-dimensional state variables.
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{The empirical problem}
\begin{itemize}\tightitems
  \item Intensities change during the day.
  \item Events are clustered and cross-exciting.
  \item Ten levels produce a high-dimensional state.
  \item Tick regime changes the meaning of the same move.
\end{itemize}
\end{block}
\end{column}
\end{columns}
\vspace{0.12em}
\begin{block}{Statistical Estimation}
\footnotesize
Approximate the conditional distribution identified by the stochastic model:
\[
f_\omega(X_\tau)\approx
\big(\mathbb{P}(y_\tau=-1\mid X_\tau),\mathbb{P}(y_\tau=0\mid X_\tau),\mathbb{P}(y_\tau=1\mid X_\tau)\big).
\]
Cross-entropy is maximum likelihood for the categorical label:
\[
\min_\omega -\sum_{\tau}\log f_\omega(X_\tau)_{y_\tau}.
\]
\end{block}
\end{frame}

\begin{frame}[plain]
\contentheader{Tick Size Sets the Scale}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.50\textwidth}
\begin{block}{Classify by spread relative to tick}
\[
\bar{\sigma}=\mathbb{E}[\sigma_\tau],
\qquad r=\frac{\bar{\sigma}}{\theta}
\]
\[
\begin{array}{ll}
\text{small-tick}  & \bar{\sigma}>3\theta,\\
\text{medium-tick} & 1.5\theta<\bar{\sigma}<3\theta,\\
\text{large-tick}  & \bar{\sigma}<1.5\theta.
\end{array}
\]
\end{block}
\end{column}
\begin{column}{0.46\textwidth}
\begin{itemize}\tightitems
  \item Large-tick stocks often have spreads near one tick, so best-level queues are central.
  \item Small-tick stocks can have wider spreads and more dispersed liquidity.
  \item The same label rule therefore creates different learning problems across stocks.
\end{itemize}
\vspace{0.25em}
\begin{block}{Scale of a Price Move}
Tick size is not only an exchange rule; it sets the economic meaning of ``one move'' in that market.
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[plain]
\contentheader{DeepLOB as a Conditional-Probability Machine}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.52\textwidth}
\includegraphics[width=\textwidth,height=0.60\textheight,keepaspectratio]{deeplob_arch_fig3.png}
\source{Zhang, Zohren and Roberts, ``Deep Learning for Limit Order Books'', Figure 3.}
\end{column}
\begin{column}{0.44\textwidth}
\begin{itemize}\tightitems
  \item Input \(X_\tau\): recent snapshots of the queueing system.
  \item CNN: local filters for bid--ask imbalance, queue depth, and depth slope.
  \item Inception: several filter widths, i.e. several local time scales.
  \item LSTM: summarizes path dependence in recent order-flow history.
\end{itemize}
\vspace{0.25em}
\begin{block}{Output}
\[
\mathrm{softmax}(h_\omega(X_\tau))_c
\approx \mathbb{P}(y_\tau=c\mid X_\tau)
\]
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[plain]
\contentheader{Why These Layers Fit the Book}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\begin{block}{Convolution and Scale}
Parameter sharing says: the same local pattern can matter at many recent times.
\[
z_{t,k}=\phi\!\left(\sum_{a,b}W_{a,b,k}X_{t+a,b}\right)
\]
Inception repeats this idea with several window widths.
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Recurrence}
The state is not Markov after compression; recent event history still matters.
\[
h_t=\Psi_\omega(z_t,h_{t-1})
\]
LSTM is a learnable memory for this missing path information.
\end{block}
\end{column}
\end{columns}
\vspace{0.35em}
\begin{block}{Why the Architecture Is Reasonable}
DeepLOB is a numerical approximation to the conditional probabilities suggested by queueing and point-process models.
\end{block}
\end{frame}

\begin{frame}[plain]
\contentheader{MCC: Correlation Instead of Raw Accuracy}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\begin{itemize}\tightitems
  \item LOB labels are often imbalanced; the stable class may dominate.
  \item Accuracy can look good by mostly predicting the majority class.
  \item MCC treats prediction and truth as correlated label variables.
  \item Random prediction is near 0; perfect prediction is 1.
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Confusion-matrix view}
\[
\text{MCC}=F(\text{all true/false positives and negatives})
\]
\vspace{0.2em}
It does not only ask how often the model is right; it asks how errors are distributed across classes.
\end{block}
\begin{block}{Interpretation}
The score measures model quality under a specific market regime, not model quality in isolation.
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[plain]
\contentheader{\(p_T\): From Labels to Trading Paths}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\begin{itemize}\tightitems
  \item F1/MCC are pointwise: each timestamp is scored separately.
  \item Trading is pathwise: entry, holding, and exit must line up.
  \item An error near an exit point is more costly than an error in a flat region.
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Complete-trade probability}
\[
\begin{aligned}
P_T&=\text{predicted trade set},\\
T_T&=\text{true trade set}.
\end{aligned}
\]
\[
p_T=\frac{|P_T\cap T_T|}{|P_T\cup T_T|}
\]
\end{block}
\begin{block}{Path-Level Meaning}
\(p_T\) asks whether the predicted sequence forms the right complete trade, not just whether isolated labels are correct.
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[plain]
\contentheader{Microstructure Explains the Scores}
\begin{enumerate}\tightitems
  \item LOB predictability is jointly determined by the model, label, horizon, and tick regime.
  \item Large-tick stocks are often easier: spreads are tight, best queues are thick, and queue changes carry more information.
  \item Classification metrics are useful but incomplete; trading signals must also be read as time-ordered paths.
  \item The paper's contribution is to put deep forecasting back inside market microstructure.
\end{enumerate}
\vspace{0.35em}
\begin{block}{Reading Rule}
Read each score together with the market structure that produced it.
\end{block}
\end{frame}

\begin{frame}[plain]
\contentheader{A Coherent Workflow}
\begin{enumerate}\tightitems
  \item Define the random object: state \(Q_t\), observation \(X_\tau\), and label \(y_\tau\).
  \item Explain the market regime: \(\bar{\sigma}/\theta\), best-queue depth, imbalance, event intensity.
  \item Choose a model bias that matches the object: local convolution, recurrence, graph structure, or point-process intensity.
  \item Evaluate two levels: sample-wise correctness and path-wise tradability.
  \item Test whether the explanation survives across stocks, horizons, and market regimes.
\end{enumerate}
\vspace{0.4em}
\begin{block}{Working Principle}
Model complexity should serve a microstructure explanation; it should not replace one.
\end{block}
\end{frame}

\begin{frame}[plain]
\contentheader{Answers to the Opening Questions}
\begin{enumerate}\tightitems
  \item What is being predicted?\\
  The future direction of the mid-price, usually after a fixed number of LOB updates:
  \[
    y_\tau=g_\theta(m_{\tau+\Delta\tau}-m_\tau).
  \]
  \item Why do tick size, spread, and horizon matter?\\
  They change queue depletion probabilities, class balance, signal lifetime, and transaction-cost meaning.
  \item Why is high accuracy not enough?\\
  A trading signal is a time-ordered path. MCC/F1 count point errors, while \(p_T\) asks whether the predicted path completes the right trade.
\end{enumerate}
\vspace{0.3em}
\begin{block}{Closing Claim}
The central question is which directional signal is predictable under this market structure, at which horizon, and whether it forms a complete trading path.
\end{block}
\end{frame}
